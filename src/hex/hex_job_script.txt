#!/bin/bash --login

#$ -cwd
#$ -l nvidia_v100=1

module load apps/binapps/anaconda3/2024.10

cp -r ~/scratch/third-year-project $TMPDIR
cd $TMPDIR/third-year-project

source $(conda info --base)/etc/profile.d/conda.sh

ENV_NAME="hex_env"

if ! conda info --envs | grep -q "^${ENV_NAME} "; then
    echo "Environment ${ENV_NAME} does not exist. Creating it..."
    conda create -n $ENV_NAME python=3.12 -y
fi 

conda activate $ENV_NAME

pip3 install numpy
pip3 install torch torchvision torchaudio
python3 -m src.alphaZero.hex.hex_zero

cp $TMPDIR/third-year-project/src/alphaZero/hex/pt_models/* ~/scratch/third-year-project/src/alphaZero/hex/pt_models

conda deactivate
